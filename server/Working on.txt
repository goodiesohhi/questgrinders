
  Meteor.startup(function() {
  var time = (new Date).toTimeString();
  var attackattempts = Meteor.user().attackattempts;

  if (Meteor.user().attacks < 10 && time = 0)
        Meteor.users.find({}).map(function(user) {
          Meteor.users.update({
            _id: user._id
          }, {
            $set: {
              'attacks': attackattempts,

            }
          })

          });
        });
